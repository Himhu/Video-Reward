```php
// 创建通用验证类
class SecurityValidator {
    const PATTERNS = [
        'email' => '/^[^\s@]+@[^\s@]+\.[^\s@]+$/',
        'phone' => '/^1[3-9]\d{9}$/',
        'username' => '/^[a-zA-Z0-9_]{3,20}$/',
        'password' => '/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)[a-zA-Z\d@$!%*?&]{8,}$/'
    ];
    
    public static function validate($input, $type, $required = true) {
        if (!$required && empty($input)) return true;
        if ($required && empty($input)) {
            throw new InvalidArgumentException("{$type} is required");
        }
        
        if (isset(self::PATTERNS[$type])) {
            return preg_match(self::PATTERNS[$type], $input);
        }
        
        return false;
    }
}
```

#### 2. XSS防护规范
```php
// 创建输出安全类
class OutputSecurity {
    public static function escape($data, $context = 'html') {
        if (is_array($data)) {
            return array_map([self::class, 'escape'], $data);
        }
        
        switch ($context) {
            case 'html':
                return htmlspecialchars($data, ENT_QUOTES, 'UTF-8');
            case 'js':
                return json_encode($data, JSON_HEX_TAG | JSON_HEX_APOS | JSON_HEX_QUOT | JSON_HEX_AMP);
            case 'url':
                return urlencode($data);
            case 'css':
                return preg_replace('/[^a-zA-Z0-9\-_.]/', '', $data);
            default:
                return htmlspecialchars($data, ENT_QUOTES, 'UTF-8');
        }
    }
}

// 在模板中使用
echo OutputSecurity::escape($user['username']); // HTML上下文
echo "var userName = " . OutputSecurity::escape($user['username'], 'js'); // JS上下文
```

```php
// ✅ 重构后 - 使用PHP内置安全函数
class PasswordSecurity {
    private const MIN_LENGTH = 8;
    private const OPTIONS = ['cost' => 12];
    
    public static function hash(string $password): string {
        if (strlen($password) < self::MIN_LENGTH) {
            throw new InvalidArgumentException('Password too short');
        }
        return password_hash($password, PASSWORD_ARGON2ID, self::OPTIONS);
    }
    
    public static function verify(string $password, string $hash): bool {
        return password_verify($password, $hash);
    }
    
    public static function needsRehash(string $hash): bool {
        return password_needs_rehash($hash, PASSWORD_ARGON2ID, self::OPTIONS);
    }
}
```

```php
// 创建配置安全管理类
class SecureConfig {
    private static $config = null;
    
    public static function get(string $key, $default = null) {
        if (self::$config === null) {
            self::loadConfig();
        }
        
        return self::$config[$key] ?? $default;
    }
    
    private static function loadConfig() {
        $configFile = ROOT_PATH . 'config/secure.php';
        if (!file_exists($configFile)) {
            throw new RuntimeException('Secure config file not found');
        }
        
        self::$config = require $configFile;
        
        // 验证必需配置项
        $required = ['database', 'jwt_secret', 'encryption_key'];
        foreach ($required as $key) {
            if (!isset(self::$config[$key])) {
                throw new RuntimeException("Missing required config: {$key}");
            }
        }
    }
}
```

### D. 访问控制和权限验证

#### 1. 统一权限验证机制
```php
// 创建权限验证中间件
class AuthMiddleware {
    public function handle($request, Closure $next, ...$permissions) {
        $user = $this->getCurrentUser($request);
        
        if (!$user) {
            return $this->unauthorized('Authentication required');
        }
        
        if (!empty($permissions) && !$this->hasPermission($user, $permissions)) {
            return $this->forbidden('Insufficient permissions');
        }
        
        return $next($request);
    }
    
    private function hasPermission($user, array $permissions): bool {
        foreach ($permissions as $permission) {
            if (!in_array($permission, $user['permissions'] ?? [])) {
                return false;
            }
        }
        return true;
    }
}
```

```php
class CSRFProtection {
    public static function generateToken(): string {
        $token = bin2hex(random_bytes(32));
        $_SESSION['csrf_token'] = $token;
        return $token;
    }
    
    public static function validateToken(string $token): bool {
        return isset($_SESSION['csrf_token']) && 
               hash_equals($_SESSION['csrf_token'], $token);
    }
    
    public static function getHiddenField(): string {
        $token = self::generateToken();
        return "<input type=\"hidden\" name=\"csrf_token\" value=\"{$token}\">";

}
```

```php
class UserController {
    private $authService;
    private $notificationService;
    private $logService;
    
    public function login(LoginRequest $request): JsonResponse {
        try {
            $credentials = $request->getValidatedData();
            $authResult = $this->authService->authenticate($credentials);
            
            if ($authResult->isSuccess()) {
                $this->notificationService->sendLoginNotification($authResult->getUser());
                $this->logService->logUserAction('login_success', $authResult->getUser());
                
                return $this->successResponse([
                    'token' => $authResult->getToken(),
                    'user' => $authResult->getUserData()
                ]);
            }
            
            return $this->errorResponse('登录失败', 401);
            
        } catch (Exception $e) {
            $this->logService->logError('login_error', $e);
            return $this->errorResponse('系统错误', 500);
        }
    }
}
```

```php
class DIContainer {
    private static $bindings = [];
    private static $instances = [];
    
    public static function bind(string $abstract, $concrete = null): void {
        self::$bindings[$abstract] = $concrete ?? $abstract;
    }
    
    public static function singleton(string $abstract, $concrete = null): void {
        self::bind($abstract, $concrete);
        // 标记为单例
    }
    
    public static function resolve(string $abstract) {
        if (isset(self::$instances[$abstract])) {
            return self::$instances[$abstract];
        }
        
        $concrete = self::$bindings[$abstract] ?? $abstract;
        
        if ($concrete instanceof Closure) {
            $instance = $concrete();
        } else {
            $instance = new $concrete();
        }
        
        self::$instances[$abstract] = $instance;
        return $instance;
    }
}
```

```php
}
```

```bash
phpmd app/ text cleancode,codesize       # 代码复杂度检查
phpcpd app/                              # 代码重复检查

# 3. 功能测试
php artisan test                         # 单元测试
php functional-test.php                  # 功能测试脚本
```

## 🎯 总结

这套重构规则涵盖了安全性、可读性、维护性三个核心维度，提供了：

1. **具体的实施标准** - 每个改进点都有明确的代码示例
2. **分阶段执行计划** - 按优先级分为4个阶段，可逐步实施  
3. **验证机制** - 包含自动化和手动验证方法
4. **风险控制** - 确保重构不影响现有功能

**建议按以下顺序开始重构：**
1. 先执行阶段一（安全基础加固）
2. 每完成一个阶段进行全面测试
3. 在生产环境上线前进行完整的回归测试

这样既能快速提升系统安全性，又能保证系统稳定性，为后续的功能扩展奠定良好基础。
